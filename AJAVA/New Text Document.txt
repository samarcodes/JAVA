import javax.swing.*;
public class progressbar
{
 public static void main(String args[])
 {
  JFrame f=new JFrame();
  JPanel p=new JPanel();
  JProgressBar b=new JProgressBar();
  b.setValue(0);
  b.setStringPainted(true);
  p.add(b);
  f.add(p);
  f.setSize(400,400);
  f.setVisible(true);
  fill();
 }
 public static void fill()
 {
  int i=0;
  while(i<100)
  {
   b.setValue(i+10);
   Thread.sleep(1000);
   i+=20;
  }
 }
}
